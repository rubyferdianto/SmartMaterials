
SPOOL c:\temp\CAX_REP_GRANT_SDB_SYS_OBJECTS.LST
SET echo on 
rem This script grants all objects FROM SDB_SYS to CAX_REP.
rem It is to run under SDB_SYS


rem granting tables
SET linesize 1000
SET HEADING OFF
SET TIMING OFF
SET TERMOUT OFF
SET PAGESIZE 0
SET ECHO off
SPOOL c:\temp\CAX_REP_GRANT_SDB_SYS_TABLES.TMP
SELECT 'GRANT SELECT ON ' || TABLE_NAME || ' to CAX_REP with grant option;'
FROM   USER_TABLES
ORDER BY TABLE_NAME;
SPOOL OFF
SET HEADING ON
SET TIMING ON
SET TERMOUT ON
SET echo on 

rem granting views
SET linesize 1000
SET HEADING OFF
SET TIMING OFF
SET TERMOUT OFF
SET PAGESIZE 0
SET ECHO off
SPOOL c:\temp\CAX_REP_GRANT_SDB_SYS_VIEWS.TMP
SELECT 'GRANT SELECT ON ' || VIEW_NAME || ' to CAX_REP with grant option;'
FROM   USER_VIEWS
ORDER BY VIEW_NAME;
SPOOL OFF
SET HEADING ON
SET TIMING ON
SET TERMOUT ON
SET echo on 

rem granting packages
SET linesize 1000
SET HEADING OFF
SET TIMING OFF
SET TERMOUT OFF
SET PAGESIZE 0
SET ECHO off
SPOOL c:\temp\CAX_REP_GRANT_SDB_SYS_PACKAGES.TMP
SELECT 'GRANT EXECUTE ON ' || OBJECT_NAME || ' to CAX_REP with grant option;'
FROM   USER_OBJECTS
WHERE  OBJECT_TYPE = 'PACKAGE'
ORDER BY OBJECT_NAME;
SPOOL OFF
SET HEADING ON
SET TIMING ON
SET TERMOUT ON
SET echo on 

SPOOL c:\temp\CAX_REP_GRANT_SDB_SYS_OBJECTS.LST

start c:\temp\CAX_REP_GRANT_SDB_SYS_TABLES.TMP
start c:\temp\CAX_REP_GRANT_SDB_SYS_VIEWS.TMP
start c:\temp\CAX_REP_GRANT_SDB_SYS_SEQUENCES.TMP
start c:\temp\CAX_REP_GRANT_SDB_SYS_PACKAGES.TMP


spool off
